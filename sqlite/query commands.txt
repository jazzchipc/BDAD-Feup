2. Qual é o conjunto de IDs dos utentes do sexo masculino com idade inferior a 30 anos que foram às urgência no dia 27 e 28 de Abril de 2016? (FEITOOOOOOO!)

SELECT Utente.numeroUtente
FROM
	(Pessoa
	INNER JOIN 
	Utente
	ON Pessoa.idPessoa = Utente.idPessoa
	INNER JOIN
	Inscricao
	ON Utente.numeroUtente = Inscricao.numeroUtente)
WHERE 
((Inscricao.horaEntrada = '27-04-2016' OR Inscricao.horaEntrada = '28-04-2016')
AND
(Pessoa.sexo == 'Masculino' AND Pessoa.idade < 30));
